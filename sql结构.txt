用户模块
	user_type		//用户类型表
		id							int				:主键id
		type_name					varchar(20)		:类型名称
		min_score					int				:达到该类型所需的最低积分
		discount					double			:折扣

		operate_time				date			:创建时间
		remark						varchar(200)	:备注

	user			//用户表
		id							int				:主键id
		user_name					varchar(20)		:用户名称
		user_phone					varchar(13)		:用户电话
		user_email					varchar(20)		:用户邮箱
		user_score					int				:用户积分	   默认：100
		user_balance				double			:用户余额
		id_card  					varchar(19)		:用户身份证
		user_birthday				date			:用户生日
		status						int				:用户是否可用	0为false,1为true,默认插入1
		operator					varchar(20)		:创建管理员
		operate_time				date			:创建时间
		remark						varchar(200)	:备注
